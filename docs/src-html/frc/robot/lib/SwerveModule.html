023/**
024 * Implements the turning and driving controllers of a swerve module.
025 *
026 * <p>The turning contoller is a combination of feedback and feedforward control. The feedback control is achieved using a
027 * profiled PID controller (trapezoidal profile and constraints) using the absolute encoder to seed the reletive encoder for
028 * feedback. The feedforward control is achieved by using a simple permanent-magnet DC motor model which zeros the
029 * acceleration coefficient (due to the trapezoid profile implementation).
030 *
031 * <p>The driving controller is also a combination of feedback and feedforward control. The feedback control is achieved
032 * using a velocity controller and the REV NEO onboard relative encoder. The feedforward control is achieved by using a
033 * simple permanent-magnet DC motor model which zeros the acceleration coefficient (due to the trapezoid profile
034 * implementation).
035 */
036public class SwerveModule {
037

039                                                    * Math.PI / Constants.Drivetrain.DRIVE_GEAR_RATIO);

098    /**
099     * Set the desired state of the swerve module.
100     *
101     * <p>The input swerve state is a speed in meters-per-second and an angle represented as a Rotation2D object. The
102     * desired angle is first optimized to keep the wheel from turning more than 90 degrees. This is done by allowing the
103     * wheel to reverse the current drive direction. Next, the turning and driving output voltages are calculated using a
104     * combination of feedback and feedforward controllers. Finally, the turning and driving output voltages are sent out to
105     * the motors.
106     *
107     * @param state the desired state of the swerve module
108     */
109    public void setDesiredState(SwerveModuleState state) {
110        m_setDesiredState = SwerveModuleState.optimize(state, new Rotation2d(getTurnAngleRad()));
111
112        m_turnPidOutput = m_turnController.calculate(getTurnAngleRad(), m_setDesiredState.angle.getRadians());
113        m_turnFeedForwardOutput = m_turnFeedForward.calculate(m_turnController.getSetpoint().velocity);
114
115        m_drivePidOutput = m_driveController.calculate(getDriveEncVelMps(), m_setDesiredState.speedMetersPerSecond);
116        m_driveFeedForwardOutput = m_driveFeedForward.calculate(state.speedMetersPerSecond);
117
118        m_turnMotor.setVoltage(m_turnPidOutput + m_turnFeedForwardOutput);
119        m_driveMotor.setVoltage(m_drivePidOutput + m_driveFeedForwardOutput);
120    }

123    /**
124     * Set offset of the modules azimuth angle by using the input absolute "home" angle and the current angle read from the
125     * duty cycle endcoder.
126     *
127     * @param homeRad the "home" angle of the duty cycle endoder in radians
128     */
129    public void setHomeOffset(double homeRad) {
130        m_homeOffset = getTurnAbsEncAngleRad() - homeRad;
131        m_turnController.reset(getTurnAngleRad());
132    }

165     *

167     */
168    public void setIdleMode(boolean isBrakeDesired) {
169        if (isBrakeDesired) {
170            m_turnMotor.setIdleMode(IdleMode.kBrake);

172        } else {
173            m_turnMotor.setIdleMode(IdleMode.kCoast);
174            m_driveMotor.setIdleMode(IdleMode.kCoast);
175        }
176    }

338    /**
339     * Constructor for a SparkMax swerve drive module using the RoboRIO to control both the steering and driving.
340     *
341     * @param label the module identifier
342     * @param turnMotorId the turning motor controller CAN ID
343     * @param driveMotorId the driving motor controller CAN ID
344     * @param pwmChannel the RoboRIO DIO channel number of the encoder PWM signal
345     * @param quadChannelA the RoboRIO DIO channel number of the encoder quadrature A signal
346     * @param quadChannelB the RoboRIO DIO channel number of the encoder quadrature B signal
347     * @param maxTurnVelRps the turning motor maximum angular velocity in radians-per-second
348     * @param maxTurnAccRpss the turning motor maximum angular acceleration in radians-per-second
349     * @param turnFeedforwardKs the turning motor FF model static coefficient
350     * @param turnFeedforwardKv the turning motor FF model velocity coefficient
351     * @param turnFeedforwardKa the turning motor FF model acceleration coefficient
352     * @param driveFeedforwardKs the driving motor FF model static coefficient
353     * @param driveFeedforwardKv the driving motor FF model velocity coefficient
354     * @param driveFeedforwardKa the driving motor FF model acceleration coefficient
355     */
356    public SwerveModule(String label, int turnMotorId, int driveMotorId, int pwmChannel, int quadChannelA, int quadChannelB,
357                        double maxTurnVelRps, double maxTurnAccRpss, double turnFeedforwardKs, double turnFeedforwardKv,
358                        double turnFeedforwardKa, double driveFeedforwardKs, double driveFeedforwardKv,
359                        double driveFeedforwardKa) {
360        m_label = label;
361
362        m_turnMotor = new CANSparkMax(turnMotorId, MotorType.kBrushless);
363        m_driveMotor = new CANSparkMax(driveMotorId, MotorType.kBrushless);
364        m_turnAbsEnc = new DutyCycleEncoder(pwmChannel);
365        m_turnRelEnc = new Encoder(quadChannelA, quadChannelB);
366        m_driveEnc = m_driveMotor.getEncoder();
367
368        m_profiledPidConstraints = new TrapezoidProfile.Constraints(maxTurnVelRps, maxTurnAccRpss);
369        m_turnController =
370            new ProfiledPIDController(Calibrations.TURN_P_GAIN, 0.0, Calibrations.TURN_D_GAIN, m_profiledPidConstraints);
371        m_driveController = new PIDController(Calibrations.DRIVE_P_GAIN, 0.0, Calibrations.DRIVE_D_GAIN);
372
373        m_turnFeedForward = new SimpleMotorFeedforward(driveFeedforwardKs, driveFeedforwardKv, driveFeedforwardKa);
374        m_driveFeedForward = new SimpleMotorFeedforward(turnFeedforwardKs, turnFeedforwardKv, turnFeedforwardKa);
375
376        m_driveMotor.restoreFactoryDefaults();
377        m_driveMotor.setSmartCurrentLimit(35);
378        m_driveMotor.enableVoltageCompensation(12.0);
379        m_driveMotor.setIdleMode(IdleMode.kBrake);
380        //m_driveMotor.burnFlash();
381
382        m_turnMotor.restoreFactoryDefaults();
383        m_turnMotor.setSmartCurrentLimit(35);
384        m_turnMotor.enableVoltageCompensation(12.0);
385        m_turnMotor.setIdleMode(IdleMode.kBrake);
386        //m_turnMotor.burnFlash();
387
388        m_driveEnc.setPositionConversionFactor(DRIVE_CONVERSION);
389        m_driveEnc.setVelocityConversionFactor(DRIVE_CONVERSION / 60.0);
390        m_turnRelEnc.setDistancePerPulse(2.0 * Math.PI / Constants.Drivetrain.turnEncPpr);
391        m_turnController.setTolerance(Units.degreesToRadians(Calibrations.MAX_TURN_ERROR_DEG));
392        m_turnController.enableContinuousInput(-Math.PI, Math.PI);
393        m_driveEnc.setPosition(0);
394
395        m_setDesiredState = new SwerveModuleState(0.0, new Rotation2d(getTurnAbsEncAngleRad()));
396        m_turnFeedForwardOutput = 0.0;
397        m_driveFeedForwardOutput = 0.0;
398        m_turnPidOutput = 0.0;
399        m_drivePidOutput = 0.0;
400
401        if (Constants.Drivetrain.ENABLE_LOGGING) {
402            m_log = DataLogManager.getLog();
403            m_turnAbsEncLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Abs Enc (rad)");
404            m_turnRelEncLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Rel Enc (rad)");
405            m_turnRelEncOffsetLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Rel Enc Offset (rad)");
406            m_turnRelEncVelocityLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Rel Enc Velocity (rad/s)");
407            m_driveEncLogEntry = new DoubleLogEntry(m_log, m_label + " Drive Rel Enc (mps)");
408            m_turnPosSetpointLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Position Setpoint (rad)");
409            m_turnVelSetpointLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Velocity Setpoint (rad/s)");
410            m_turnPosErrorLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Position Error (rad)");
411            m_turnVelErrorLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Velocity Error (rad/s)");
412            m_turnFeedforwardOutLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Feed-forward Output (V)");
413            m_turnPidOutLogEntry = new DoubleLogEntry(m_log, m_label + " Turn PID Output (V)");
414            m_turnApplidedVoltageLogEntry = new DoubleLogEntry(m_log, m_label + " Turn Applied Voltage (V)");
415        } else {
416            m_log = null;
417            m_turnAbsEncLogEntry = null;
418            m_turnRelEncLogEntry = null;
419            m_turnRelEncOffsetLogEntry = null;
420            m_turnRelEncVelocityLogEntry = null;
421            m_driveEncLogEntry = null;
422            m_turnPosSetpointLogEntry = null;
423            m_turnVelSetpointLogEntry = null;
424            m_turnPosErrorLogEntry = null;
425            m_turnVelErrorLogEntry = null;
426            m_turnFeedforwardOutLogEntry = null;
427            m_turnPidOutLogEntry = null;
428            m_turnApplidedVoltageLogEntry = null;
429        }
430    }
