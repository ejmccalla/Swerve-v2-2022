001package frc.robot.commands;
002
003import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
004import edu.wpi.first.wpilibj2.command.CommandBase;
005import frc.robot.subsystems.Drivetrain;
006import frc.robot.subsystems.Drivetrain.StateType;
007
008/**
009 * Implements a command to run to help with calibrating the wheel diameter.
010 * {@link frc.robot.Calibrations#WHEEL_DIAMETER_INCH}.
011 */
012public class CalibrateWheelDiameter extends CommandBase {
013
014    private Drivetrain m_drivetrain;
015
016    /**
017     * Constructor for the calibrate wheel diameter command.
018     *
019     * @param drivetrain uses the drivetrain for drive encoder feedback to the smart dashboard
020     */
021    public CalibrateWheelDiameter(Drivetrain drivetrain) {
022        m_drivetrain = drivetrain;
023        addRequirements(m_drivetrain);
024    }
025
026    /**
027     * Set the swerve modules to coast mode (since the robot will manually be pushed) and reset the drive encoders to 0.
028     */
029    @Override
030    public void initialize() {
031        m_drivetrain.updateState(StateType.Calibrating);
032        m_drivetrain.resetModulesDriveEncoder();
033    }
034
035    /**
036     * Set the swerve modules motor ouptuts to 0 volts. The drive encoder positions are output to the smart dashboard for 
037     * recording.
038     */
039    @Override
040    public void execute() {
041        m_drivetrain.setIdleModules();
042        SmartDashboard.putNumberArray("Wheel Calibration Rotations", m_drivetrain.getModulesDriveRotations());
043    }
044
045    /**
046     * Update the drivetrain state when the command is finished.
047     */
048    @Override
049    public void end(boolean interrupted) {
050        m_drivetrain.setModulesIdleMode(false);
051        m_drivetrain.setIdleModules();
052        m_drivetrain.updateState(StateType.Idle);
053
054    }
055
056}
